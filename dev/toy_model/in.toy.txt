#Input File for a simple diffusion model
echo both
seed 9875
app_style diffusion linear hop
dimension 2
lattice sq/8n 1.0
variable n_rep equal 100
region myregion block 0 ${n_rep} 0 ${n_rep} 0 1
create_box myregion
create_sites box
set site value 2
variable V_num equal 10
variable V_frac equal (${V_num}/((${n_rep}^2)))
set site value 1 fraction ${V_frac}  

barrier hop 1 0.69

temperature 0.04
solve_style linear

dump dump1 text 0.25 dump.out
stats 0.1

variable i loop 10
label loop
run 1
next i
